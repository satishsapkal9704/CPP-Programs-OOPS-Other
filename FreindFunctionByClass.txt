#include <QCoreApplication>
#include<iostream>
using namespace std;
class Animal
{
private:
    string name;
    int age;
public:
    Animal()
    {
         name="Satish";
         age=22;
    }
public:
   friend class Dog;
   friend class Cat;
};
class Dog
{
public:
    void disp(Animal & a)
    {
        cout<<"name and sage are"<<a.name<<a.age<<endl;
    }
};
class Cat{
public:
    void fats(Animal & a2)
    {
        cout<<"name and age are"<<a2.name<<a2.age<<endl;
    }
};

int main()
{
   Animal a1;
   Animal a3;
   Dog d1;
   Cat c1;
   d1.disp(a1);
   c1.fats(a3);
  return 0;

};
