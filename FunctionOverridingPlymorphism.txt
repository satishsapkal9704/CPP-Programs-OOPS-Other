#include <QCoreApplication>
#include<iostream>
using namespace std;
class Animal
{
public:
    virtual void eat()
    {
        cout<<"adding"<<endl;
    }
};

class Dog:public Animal
{
public:
    void eat() override
    {
        cout<<"override here"<<endl;
    }
    void bark()
    {
        cout<<"barking"<<endl;
    }
};
int main()
{
   Dog dog;
   dog.eat();
   dog.bark();
   return 0;
};
