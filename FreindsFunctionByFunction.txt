#include <QCoreApplication>
#include<iostream>
using namespace std;
class Animal
{
private:
    string name;
    int age;
public:
    Animal()
    {
         name="Satish";
         age=22;

    }
public:
    friend void disp(Animal & a);
    friend void disp2(Animal & a3);
};

void disp(Animal & a)
{
    cout<<"namwe and ag is"<<a.name<<a.age<<endl;
}
void disp2(Animal & a3)
{
    cout<<"name and age are as follows"<<a3.name<<a3.age<<endl;
}
int main()
{
    Animal a2;
    disp(a2);
    Animal a4;
    disp2(a4);
    return 0;
};
